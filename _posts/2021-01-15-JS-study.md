## Object
기본적으로 key-value 구조


```
const obj = { 
    x: 3,
    y: 4,
    z: 5
}

const obj2 = {
    "x": 3,
    "y": 4,
    "z": 5
}

// obj.x 가능
// obj[x] 불가능 obj["x"] 가능
// obj2.x 가능 obj2."x" 불가능
// obj2["x"] 가능

for (key in obj)
    console.log(key, obj[key])

/*
for (key in obj)
    console.log(key, obj.key)
   */

const obj3 = {
    3: "x", 
    4: "y",
    5: "z"
}
// obj3["3"], obj3[3] 가능
// obj3.3 불가능

```
